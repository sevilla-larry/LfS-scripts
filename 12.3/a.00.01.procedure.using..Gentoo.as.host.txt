        set default=0
        set timeout=5

        insmod part_gpt
        insmod ext2
        set root=(hd0,2)

        insmod efi_gop
        insmod efi_uga
        if loadfont /boot/grub/fonts/unicode.pf2; then
          terminal_output gfxterm
        fi

        menuentry "GNU/Linux, Linux 6.10.5-lfs-12.3-r00" {
          linux   /boot/vmlinuz-6.10.5-lfs-12.3-r00 root=/dev/vda2 ro   # Qemu/KVM VM
